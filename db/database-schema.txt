=====================
DATABASE ORGANIZATION
=====================
Subjects Database: One database. Contains Publisher, Standard, Subject data. Example: subjects.json
Concepts Database: One database per subject. Contains Chapter, Topic, SubTopic, Content data. Example: concepts_6_science.json
QA Database: One database per subject. Contains Questions, Answers data. Example: qa_6_science.json

================================
SUBJECTS DATABSE (SUBJECTS.JSON)
================================
Publisher = {
    "id": "string", // Random 8 character alpha-numeric string
    "publisherName": "string",
    "standards": [ Standard ]
}

Standard = {
    "id": "string", // Random 8 character alpha-numeric string
    "standardName": "string",
    "subjects": [ Subject ]
}

Subject = {
    "id": "string", // Random 8 character alpha-numeric string
    "subjectName": "string"
}

==========================
CONCEPT DB (CONCEPT.JSON)
==========================
Chapter = {
    "id": "string", // Random 8 character alpha-numeric string
    "subjectId": "string", // parent subject id
    "chapterNo": "integer", // chapter number
    "chapterName": "string", // chapter name
    "topics": [ Topic ] // topics
}

Topic = {
    "id": "string", // Random 8 character alpha-numeric string
    "title": "string", // topic title
    "subTopics"?: [ SubTopic ] // subtopics
}

SubTopic = {
    "id": "string", // Random 8 character alpha-numeric string
    "title": "string", // subtopic title
    "content"?: [ Content ] // content
}

Content = {
    "id": "string", // Random 8 character alpha-numeric string
    "type": "body|bullet1|bullet2|number1|number2|image|video|equation", // type of content
    "text"?: "string", // for body/bullet1/bullet2/number1/number2
    "equation"?: "string", // for equation
    "url"?: "string", // for image/video
    "metadata": {  // For image/video
        "altText": "string",
        "size": "string" // small/medium/large/very large
    }
}

===============
QA DB (QA.JSON)
===============
Chapter = {
    "id": "string", // Random 8 character alpha-numeric string
    "subjectId": "string", // parent subject id
    "chapterNo": "string", // chapter number
    "chapterName": "string", // chapter name
    "qa": [ Qa ] // questions and answers
}

Qa = {
    "id": "string", // Random 8 character alpha-numeric string
    "exerciseType": "book|board|extra", // type of exercise
    "questionType": "short|long|very-short", // type of question
    "reference": "string", // reference
    "difficulty": "easy|medium|hard", // difficulty level
    "mcqAnswer"?: "integer", // mcq answer
    "question": [ Content ], // question
    "answer": [ Content ] // answer
}

Content = {
    "id": "string", // Random 8 character alpha-numeric string
    "type": "plain|bullet1|bullet2|number1|number2|image|video|equation", // type of content
    "text"?: "string", // for bullet1/bullet2/number1/number2
    "equation"?: "string", // for equation
    "url"?: "string", // for image/video
    "metadata": {  // For image/video
        "altText": "string",
        "size": "string", // small/medium/large/very large
    }
}
